Step 2.2 — The wiring pattern

At the top of the <script> block, replace all simulated data with:

javascript
let DATA = null;
let currentModel = 'model_2009'; // A/B toggle state

async function loadData() {
  const res = await fetch('data/fragility_output.json');
  DATA = await res.json();
  init(); // render everything after data loads
}

function getScores() {
  return DATA.models[currentModel].monthly_scores;
}

function getScore(dateStr) {
  const scores = getScores();
  const entry = scores.find(s => s.date.startsWith(dateStr));
  return entry ? entry.fragility_score : null;
}

function getRegime(dateStr) {
  const entry = getScores().find(s => s.date.startsWith(dateStr));
  return entry ? entry.regime : 'HEDGE';
}

function getShap() {
  return DATA.models[currentModel].shap_values;
}

function getCorrelations(dateStr) {
  const entry = getScores().find(s => s.date.startsWith(dateStr));
  return entry ? entry.correlations : {};
}

document.addEventListener('DOMContentLoaded', loadData);

Then every chart rendering function that currently calls monthToScoreA(date) becomes getScore(date), correlation heatmap calls getCorrelations(date), SHAP bar chart calls getShap().

Export functionality

Every figure on the dashboard should have a PNG download button — one <canvas>.toBlob() call per chart. For the report write-up, Vansh/Sahar will just click download and attach the plot directly.

javascript
function addDownloadButton(chartCanvas, filename) {
  const btn = document.createElement('button');
  btn.textContent = '↓ Export PNG';
  btn.onclick = () => {
    chartCanvas.toBlob(blob => {
      const a = document.createElement('a');
      a.href = URL.createObjectURL(blob);
      a.download = filename;
      a.click();
    });
  };
  chartCanvas.parentNode.appendChild(btn);
}
